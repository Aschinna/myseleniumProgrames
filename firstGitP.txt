	@Parameters({"browse"})
	@BeforeMethod
	public void beforeMethod(String browse ) {
		switch(browse.toLowerCase()) {
		case "chrome":
			driver = new ChromeDriver();
			break;
		case "firefox":
			driver = new FirefoxDriver();
			break;
		default:

			System.out.println("Enter valid Browser Name");
			break;
		}
		driver.manage().window().maximize();
	}
